trait A {
    test() {
        print "I'm A";
    }
}

trait B {
    test1() {
            print "I'm B";
        }
}

trait Compose with A, B {
    test3() {
        this.test();
        this.test1();
    }
}

trait Test1 {
    abc() {
        print "abc";
    }
}

class C with Compose, Test1 {}

var c = C();
c.test();
c.test3();
c.abc();
